# 安装Nodejs
//安装cnpm
npm install -g cnpm --registry=https://registry.npm.taobao.org
//升级npm和cnpm
npm install -g npm
cnpm install -g cnpm
//查看已经安装的模块
cnpm list
//在项目目录中安装vue
cnpm i vue --save
//进入node模式
node

# vue多页面应用文件引用
//在linux命令行中安装
cnpm i vue --save
//或者引用链接
<script src="https://cdn.jsdelivr.net/npm/vue@2.5.17/dist/vue.min.js"></script>

# 安装vue-cli
npm install -g vue-cli
